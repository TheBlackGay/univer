const getResultFromFormula = (formulaResult: Nullable<ICellData>[][]) => {
    const v = formulaResult && formulaResult[0] && formulaResult[0][0];
    if (v?.t === CellValueType.BOOLEAN) {
        return v.v === BooleanNumber.TRUE || v.v === true;
    }
    return v ? v.v : false;
};
